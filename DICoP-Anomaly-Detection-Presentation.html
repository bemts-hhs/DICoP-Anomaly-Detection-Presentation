# load trauma registry counts and differences 
trauma_counts = DataFrame(
  XLSX.readtable(iowa_trauma_registry_counts_path, "data")
  );

# load trauma registry anomalies
trauma_anomalies = DataFrame(
  XLSX.readtable(iowa_trauma_registry_anomalies_path, "anomalies")
  );

# summarize anomalies identified using each strategy for the trauma use case
trauma_anomalies_summary = @chain trauma_counts begin
  @summarize(z_anomaly_total = sum(z_anomaly_2025),
             z_anomaly_only = sum(z_anomaly_2025 & !pct_anomaly_2025 & !nb_pois_anomaly_2025),
             pct_anomaly_total = sum(pct_anomaly_2025),
             pct_anomaly_only = sum(pct_anomaly_2025 & !z_anomaly_2025 & !nb_pois_anomaly_2025),
             nb_anomaly_total = sum(nb_pois_anomaly_2025),
             nb_anomaly_only = sum(nb_pois_anomaly_2025 & !z_anomaly_2025 & !pct_anomaly_2025),
             total_anomalies = sum(z_anomaly_2025 | pct_anomaly_2025 | nb_pois_anomaly_2025),
             total_facilities = n()
             )
  @pivot_longer(
    [z_anomaly_total, z_anomaly_only, pct_anomaly_total, pct_anomaly_only, nb_anomaly_total, nb_anomaly_only], 
    names_to = anomaly_method, 
    values_to = anomaly_count
    )
  @relocate(
    [total_anomalies, total_facilities],
    after = anomaly_count
  )
  @mutate(
    anomaly_pct = round(anomaly_count / total_anomalies * 100; digits=2),
    facility_pct = round(anomaly_count / total_facilities * 100; digits=2),
    overall_anomaly_pct = round(total_anomalies / total_facilities * 100; digits=2),
    anomaly_method = replace.(anomaly_method, r"_total" => "")
  )
end

Introduction

Project Purpose

  • Annual anomaly detection for EMS agencies and trauma centers.
  • Use of distributional methods and practical thresholds.
  • Overview of data, methodology, and results.

Data Description

Source and Structure

  • Annual counts of record submission (2020–2026).
  • Stable data generating process via scheduled reports from the registries.

Distributional Characteristics

  • Annual EMS and trauma registry counts follow discrete, right‑skewed distributions.
  • We use probability distributions to characterize expected reporting patterns at the facility or agency level.
  • Emphasis is on the marginal empirical distributions for each reporting entity, not population parameters.

Probability Distributions

Important distributions for these data:

  • Normal
  • Poisson
  • Negative Binomial
  • Why quasi-Poisson was not used.

The Normal {\(\text{N}(\mu, \sigma^2)\)} Distribution

  • The normal distribution is shown for reference when evaluating year‑to‑year differences.
  • The Poisson distribution represents equidispersed count processes.
  • The quasi‑Poisson example demonstrates increased variability due to Gamma–Poisson mixing.
  • The Negative Binomial distribution provides strong right‑tail behavior consistent with overdispersed annual record counts.

Anomaly Detection Methods

Z‑Score

  • Year-over-year differences.
  • Approximate normality.

Percent Change

  • Operational thresholding.

NB–Poisson Prediction Interval

  • Distribution-based expected ranges.

Results

Summary of Findings

  • Facilities flagged by each detector.
  • Agreement between detectors.
